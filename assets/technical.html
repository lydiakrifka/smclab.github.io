3. Example Code Snippet

Minimal baseline for feature extraction and normalization (used in several experiments): 

# features.py – tiny baseline pipeline
                import librosa, numpy as np
                from sklearn.preprocessing import StandardScaler

                def extract_log_mel(path, n_mels=64):
                y, sr = librosa.load(path, sr=None)
                mel = librosa.feature.melspectrogram(y=y, sr=sr, n_mels=n_mels)
                log_mel = librosa.power_to_db(mel, ref=np.max)
                scaler = StandardScaler()
                X = scaler.fit_transform(log_mel.T) # frames × features
                return X, sr

                X, sr = extract_log_mel("audio/example.wav")
                print(X.shape, "sample rate:", sr)
